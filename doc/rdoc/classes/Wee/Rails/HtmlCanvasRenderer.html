    def build_url(hash={})
      cid = hash[:callback_id]
      url = rendering_context.controller.url_for :action => 'callback'
      url << "?__c=#{ rendering_context.component_name }"
      url << "&__a=#{ rendering_context.redirect_action }" if rendering_context.redirect_action 
      url << "&#{ cid }" if cid
      url
    end

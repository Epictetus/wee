Class Wee::HtmlWriter
In: lib/wee/renderer/html/writer.rb
Parent: Object

A class used to write out HTML documents easily.

Usage: 

  w = Wee::HtmlWriter.new(doc='')
  w.start_tag('html')
  w.start_tag('body')
  w.start_tag('a', 'href' => 'http://...')
  w.text('link')
  w.end_tag('a')
  w.end_tag('body')
  w.end_tag('html')

  p doc        # => '<html><body><a href="http://...">link</a></body></html>'

  def single_tag(tag, attributes=nil)
    if attributes
      @port << "<#{ tag }"
      attributes.each {|k, v| 
        if v
          @port << %[ #{ k }="#{ v }"] 
        else
          @port << %[ #{ k }] 
        end
      }
      @port << " />"
    else
      @port << "<#{ tag } />"
    end

    self
  end

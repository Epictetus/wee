  def to_callback(symbol, args, block)
    raise ArgumentError if symbol and block
    if symbol
      Wee::LiteralMethodCallback.new(@canvas.current_component, symbol, *args)
    else
      raise ArgumentError if not args.empty?
      block
    end
  end

    def with(component)
      rclass = component.renderer_class
      if rclass == self
        # reuse renderer
        old_component = @current_component 
        begin
          @current_component = component
          yield self
        ensure
          @current_component = old_component
        end
      else
        close
        r = rclass.new(@request, @response, @callbacks, @document, component)
        begin
          yield r
        ensure
          r.close
        end
      end
    end
